    $(document).ready(function(){

    function filterSubmit () {
        if(_tracking == 1) {
            var opt_label = '';
            var opt_value = '';        

            if($("input[name='stars[]']:checked").length) {
                opt_label += "Xếp hạng sao" + " + ";
                $.each($("input[name='stars[]']:checked"), function() {   
                    opt_value += $(this).next().text() + " + ";
                });
            }

            if($("input[name='prices[]']:checked").length) {
                opt_label += "Theo giá phòng" + " + ";
                $.each($("input[name='prices[]']:checked"), function() {                 
                    opt_value += $(this).next().text() + " + ";
                }); 
            }

            if($("input[name='persons[]']:checked").length) {
                opt_label += "Số khách / phòng" + " + ";
                $.each($("input[name='persons[]']:checked"), function() {                 
                    opt_value += $(this).next().text() + " + ";
                }); 
            }

            if($("input[name='facilities[]']:checked").length) {
                opt_label += "Tiện nghi khách sạn" + " + ";
                $.each($("input[name='facilities[]']:checked"), function() {                 
                    opt_value += $(this).next().text() + " + ";
                }); 
            }

            if($("input[name='geo_near[]']:checked").length) {
                opt_label += "Khu vực" + " + ";
                $.each($("input[name='geo_near[]']:checked"), function() {                 
                    opt_value += $(this).next().text() + " + ";
                }); 
            }

            opt_label = Trim(opt_label, " + ");
            opt_value = Trim(opt_value, " + ");
            opt_label = opt_label + " / " + opt_value;

            // Prevent the form being submitted just yet
            //e.preventDefault();
            // Keep a reference to this dom element for the callback
//            var _this = this;
            _gaq.push(
                // Queue the tracking event
                ['_trackEvent', 'Hotel listing', 'Filter hotel', opt_label],
                // Queue the callback function immediately after.
                // This will execute in order.
                function() {
                    // Submit the parent form
                    $('#hsearch').submit();
                }
            );
        } else {
            $('#hsearch').submit();
        }
    }
</script>
